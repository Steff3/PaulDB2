001    package grails.ldap;
002    
003    import org.codehaus.groovy.grails.commons.*;
004    import gldapo.schema.annotation.GldapoNamingAttribute;
005    import java.lang.reflect.Field;
006    
007    public class LdapClassArtefactHandler extends ArtefactHandlerAdapter {
008    
009        public LdapClassArtefactHandler() {
010            super("Ldap", GrailsLdapClass.class, DefaultGrailsLdapClass.class, null);
011        }
012    
013        public GrailsClass newArtefactClass(Class artefactClass) {
014            return new DefaultGrailsLdapClass(artefactClass);
015        }
016    
017        public boolean isArtefactClass(Class clazz) {
018            return isLdapClass(clazz);
019        }
020    
021        public static boolean isLdapClass(Class clazz) {
022            if (clazz == null) return false;
023            for (Field field : clazz.getDeclaredFields()) {
024                if (field.getAnnotation(GldapoNamingAttribute.class) != null)
025                    return true;
026            }
027            return false;
028        }
029    }
